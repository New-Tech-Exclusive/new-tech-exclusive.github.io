const express = require('express'); const axios = require('axios'); const cors = require('cors'); const { URL } = require('url'); const app = express(); const PORT = process.env.PORT || 3000; // Enable CORS for all origins app.use(cors()); app.use(express.json()); app.use(express.urlencoded({ extended: true })); // Serve a simple HTML interface app.get('/', (req, res) => { res.send(` Web Proxy Server

🌐 Web Proxy Server

How to use:
  • Enter any website URL above
  • Click "Load Website" to access it through the proxy
  • The proxy will bypass most network restrictions
API Endpoint: GET /proxy?url=YOUR_URL
`); }); // Main proxy endpoint app.get('/proxy', async (req, res) => { try { const targetUrl = req.query.url; if (!targetUrl) { return res.status(400).send('URL parameter is required'); } // Validate URL let parsedUrl; try { parsedUrl = new URL(targetUrl); } catch (e) { return res.status(400).send('Invalid URL provided'); } // Make request to target URL with headers that bypass restrictions const response = await axios.get(targetUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36', 'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8', 'Accept-Language': 'en-US,en;q=0.5', 'Accept-Encoding': 'gzip, deflate, br', 'Connection': 'keep-alive', 'Upgrade-Insecure-Requests': '1' }, maxRedirects: 5, validateStatus: function (status) { return status < 500; // Accept any status code less than 500 }, responseType: 'arraybuffer' }); // Forward response headers const contentType = response.headers['content-type']; if (contentType) { res.setHeader('Content-Type', contentType); } // Modify HTML content to fix relative URLs if (contentType && contentType.includes('text/html')) { let html = response.data.toString('utf-8'); // Convert relative URLs to absolute const baseUrl = `${parsedUrl.protocol}//${parsedUrl.host}`; html = html.replace(/href="\/([^"]*)"/g, `href="${baseUrl}/$1"`); html = html.replace(/src="\/([^"]*)"/g, `src="${baseUrl}/$1"`); // Inject base tag html = html.replace(//i, ``); res.send(html); } else { res.send(response.data); } } catch (error) { console.error('Proxy error:', error.message); res.status(500).send(`

Error loading website

${error.message}

The website may be blocking proxy access or may be unavailable.

`); } }); // API endpoint for programmatic access app.post('/api/proxy', async (req, res) => { try { const { url, method = 'GET', headers = {}, data } = req.body; if (!url) { return res.status(400).json({ error: 'URL is required' }); } const response = await axios({ method, url, headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', ...headers }, data, maxRedirects: 5, validateStatus: () => true }); res.json({ status: response.status, headers: response.headers, data: response.data }); } catch (error) { res.status(500).json({ error: error.message }); } }); app.listen(PORT, () => { console.log(`Proxy server running on http://localhost:${PORT}`); console.log(`Use: http://localhost:${PORT}/proxy?url=WEBSITE_URL`); });
